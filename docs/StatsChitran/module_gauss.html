13    """
14    This function generates a list of numbers that follows a gaussian distribution
15    :param list float x: A floating list of the numbers over which a probability or statistical distribution is expected
16    :param amp: The amplitude of the gaussian, only expected if probability is false(Statistical distribution and not a probability distribution). Providing this value when probability is True throws an error
17    :param mu: The mean of the gaussian. Defaults to zero
18    :param sig: The standard deviation of the gaussian, defaults to 1
19    :param probability: True for a probability distribution, False for a statistical distribution. Defaults to False.
20    :return: list[float]
21    :Example:
22    >>> #import the necessary libraries
23    >>> import numpy as np
24    >>> import matplotlib.pyplot as plt
25    >>> #Build the x-values
26    >>> np.linspace(-7, 7, 100000)
27    >>> x = x.tolist()
28    >>> #Build two gaussians
29    >>> y1 = gauss(x, sig = 3, probability=True)
30    >>> y2 = gauss(x, mu = 1, sig = 1.5, probability=True)
31    >>> #Plot the two gaussians
32    >>> plt.plot(x, y1, color = 'blue', label = 'g1')
33    >>> plt.plot(x, y2, color = 'red', label = 'g2')
34    >>> plt.xlabel('X-axis Label')
35    >>> plt.ylabel('Y-axis Label')
36    >>> plt.show()
37        Here is a plot of the input(x) versus outputs(y1 and y2) lists:
38
39    ![Input vs Output List](../images/module_gauss.png)
40    """
41    ##Error checking. Can't have a probability distribution and an amplitude both assigned
42    if( ( probability == True ) and ( amp != None) ):
43        raise ValueError('amp cant be manually assigned when probability is True')
44    elif ( (probability == False) and (amp == None) ):
45        raise ValueError('amp has to be specified when probability is False')
46    elif (probability == True):
47        amp = 1/(sig*m.sqrt(2*Constants.PI))
48    else:
49        pass
50    x = np.array(x)
51    y = amp*np.exp(-1*( ( x - mu )/(m.sqrt(2)*sig) )**2 )
52    y = y.tolist()
53    return y
