13    :param v2: the second point in a list of the sc.Vec type. Needs to be a 2 element list.

17    Examples

40    """
41    # v1 and v2 have to be two 2D point vectors(numeric) of type sc.Vec
42    if ( len(Vec.plist(v1)) != 2 ) | ( len(Vec.plist(v2)) != 2 ):
43        raise ValueError('v1 and v2 both have to be list type sc.Vec objects of lengths equal to 2')
44    elif isinstance(v1, sc.Vec) and isinstance(v2, sc.Vec) and isinstance(x, sc.Vec) == False:
45        raise ValueError('v1, v2 and x have to be list type sc.Vec objects')
46    elif is_numeric(Vec.plist(v1)) and is_numeric(Vec.plist(v2)) and is_numeric(Vec.plist(x))== False:
47        raise ValueError('v1, v2 and x have to be numeric objects converted to sc.Vec objects')
48    else:
49        x1 = v1[1]; x2 = v2[1]; y1 = v1[2]; y2 = v2[2]
50        m = (y2 - y1)/(x2 - x1)

52        y = Vec.scale(x, m)
53        y = Vec.shift(y, c)
54        return y
